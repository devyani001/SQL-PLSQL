Finding Highesh paid teacher teaching students
--------------------------------------------------
Solution-1
-----------
SELECT student.name
FROM student, teacher, st
WHERE student.rno =st.rno AND teacher.tno = st.tno
AND salary = (select max(salary) from teacher);

Solution-2
-----------
select * 
from student 
where rno IN (
				select rno 
				from st 
				where tno IN(
								select tno 
								from teacher 
								where salary = (select max(salary) from teacher)
			 );

SELECT teacher.name AS tname, student.name as sname 
FROM teacher, student 
WHERE teacher.tno = student.tno 
ORDER BY teacher.name;

Using Join-
-----------
SELECT teacher.name AS tname, student.name as sname 
FROM teacher JOIN student ON teacher.tno = student.tno 
ORDER BY teacher.name;
